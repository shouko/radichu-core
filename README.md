# Radichu Core

Core of Radichu

## Usage

```js
const radichuCore = require('radichu-core');

radichuCore.configure({
  "apiEndpoint": "https://example.jp/v2/api",
  "metadataEndpoint": "https://example.jp/v3",
  "headerPrefix": "X-MyAwesome-App-",
  "appName": "aFeaturePhone7a",
  "fullKey": "EXAMPLEKEY=="
});
```
